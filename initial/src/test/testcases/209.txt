
        class Shape {
            float length, width;
            float getArea() {}
            Shape(float length, width){
                this.length := length;
                this.width := width;
            }
        }

        class Rectangle extends Shape {
            float getArea(){
                return this.length*this.width;
            }
        }
        class Triangle extends Shape {
            float getArea(){
                return this.length*this.width / 2;
            }
        }
